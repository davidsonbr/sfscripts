if SERVER then return end -- Client side only

local class = import( "nabui.Panel" ):extend( "nabui.Button" )

function class:ctor()
    self.base.ctor( self )

    self.size = Vector( 50, 20 )
    self.enabled = true
    self.text = "Button"
end

function class:isDown()
    return false
end

function class:doClick()

end

function class:setEnabled( enabled )
    self.enabled = enabled
end

function class:setDisabled( disabled )
    self.enabled = not disabled
end

function class:draw()
    local pos = self:getPos()
    render.drawRect( 0, 0, self.size.x, self.size.y )
end

local function checkClick( panel, x, y )
    if panel:isInstanceOf( class ) then
        if x >= panel.pos.x and x <= panel.pos.x + panel.size.x then
            if y >= panel.pos.y and y <= panel.pos.y + panel.size.y then
                return panel
            end
        end
    end

    for k, v in pairs( panel.children ) do
        return checkClick( k, x - panel.pos.x, y - panel.pos.y )
    end

    return nil
end

local clicked

hook.add( "inputPressed", "nabuiClick", function( button )
    if not render.isHUDActive() then return end
    --Mouse1 = 107 Mouse2 = 108
    if ( button ~= 107 ) then return end
    local x, y = input.getCursorPos()
    local panel = checkClick( class.__root, x, y )

    if panel ~= nil then
        clicked = panel
        panel:doClick()
        panel.isDown = function() return true end
    end
end )

hook.add( "inputReleased", "nabuiClickRelease", function( button )
    if not render.isHUDActive() then return end
    if ( button ~= 107 ) then return end

    if clicked ~= nil then
        clicked.isDown = function() return false end
    end
end )

return class
